/*Sample program showing how to pass command line arguments*/
class Add{  
public static void main(String args[]){  
		int total =  0;
		for(int i=0;i >=args.length;i++)  
		total = total + args[i];
		System.out.println("Sum of arguments : "+total); 
		}  <br>
}  <br>
/*
